// Toggle
F3			= Toggle Debug
E			= Toggle Mouse Lock
+			= Add Model
ENTER		= List Manager
Ctrl + R	= Reset Camera
F2			= Take Screenshot
ESCAPE		= Toggle Pause

// Movement:
W			= Forward
S			= Backward
D			= Right
A			= Left
TAB			= Move Faster
SPACE		= Up
SHIFT		= Down

// Mouse Lock:
MOUSE		= Rotate Camera X, Y
SCROLLWHEEL	= Rotate Camera Z
